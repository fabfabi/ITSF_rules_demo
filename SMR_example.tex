\newcommand{\gtterm}[2]{%Initial definition for the glossary
%	{\defformat{#1}: #2}
	\glstarget{##1}{
		\newglossaryentry{#1}{
				name=#1,
				description = {#2}
			}
		}
}

\newcommand{\dtterm}[1]{
%{\defformat{#1}: #2}
\begin{Definition}[name=\defformat{#1}]
	\label{d:#1}
	 \glsdesc{#1}
\end{Definition}
}
