%\newcommand{\tterm}[2]{{\defformat{#1}: #2}} % this one works :-)

\newcommand{\gtterm}[2]{%Initial definition for the glossary
%	{\defformat{#1}: #2}
	\newglossaryentry{#1}{
			name=#1,
			description = {#2}
		}
}

\newcommand{\dtterm}[1]{
%{\defformat{#1}: #2}
\begin{Definition}[name=\defformat{#1}]
%\label{#1}
	 \glsdesc{#1}
\end{Definition}
}

\newcommand{\tref}[1]{\refformat{\gls{#1}}}

\newcommand{\tterm}[1]{
	\defformat{#1}: \glsdesc{#1}
}
